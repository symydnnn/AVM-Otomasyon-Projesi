						printf(" %d  ",matris[i][j]);

				}

			for(i=0;i<50;i++){ printf("_");} printf("\n");
			printf("\n");

			printf("Haftalara göre günlük yoğunluk ortalaması:\n");

				for(j=0;j<7;j++){
					gunlukortHafta+=(float)(matris[i][j]);
				}
			gunlukortHafta=gunlukortHafta/7;
			gunlukortHafta=gunlukortHafta/10;
			printf("%d) hafta :  %%%.2f  ",(i+1), gunlukortHafta);
			printf("\n");

				for(j=0;j<7;j++){
					gunlukortaHepsi+=(float)(matris[i][j]);	
				}	
			}

			gunlukortaHepsi=( gunlukortaHepsi/(k*7) );
			gunlukortaHepsi=gunlukortaHepsi/10;
			printf("Genel ortalama: %%%.2f\n",gunlukortaHepsi);
			
			for(i=0;i<50;i++){ printf("_");} printf("\n");
	
			printf("\n\nÇıkmak için E, yeniden bakmak için H giriniz: ");
			fflush(stdin);

			}
			else if(toupper(getch())== 'E'){
				printf("\n\n");
			}
			sleep(1);

	    }
	}
}

						printf("[1] Geri dönmek için\n[2] Ana Menüye Çıkmak için\nGiriniz: ");
						scanf("%d",&arasecim);
						if(arasecim==1) goto x;
						else if(arasecim) AnaMenu();}
					else{
						printf("\nKaydınız oluşturulmuştur fakat mevcut bir mağazanız bulunmamaktadır\nMağaza Kiralama menüsüne yönlendiriliyorsunuz...\n");
						fclose(magaza);
						sleep(3);
						MagazaKiralama();}
					break;
				case 2:
					if(id==NULL){
						printf("\nHenüz tarafınızdan ışıklandırmanız belirlenmemiştir.\nIşıklandırma Menüsüne Yönlendiriliyorsunuz...\n\n");
						fclose(magaza);
						sleep(1);
						isiklandirma();}
					else{
						fread(&Boyut,sizeof(Boyut),1,id);
						if(Boyut.saatler.aydinlatmaderecesi>0){
							printf("\nMağazanın aydınlatması yüzde %d olarak ayarlanmıştır.\n\n",Boyut.saatler.aydinlatmaderecesi);
							for(i=0;i<100;i++){printf("_");}printf("\n");
							printf("[1] Değişiklik için\n[2] Çıkış yapmak için\nGiriniz: ");
							scanf("%d",&secim);
								if(secim==1){
									fclose(magaza);
									isiklandirma();}
							}
						else{
							printf("Mağaza kapalıdır.\n[1] Açılış için\n[2] Çıkış yapmak için\nGiriniz: ");
							scanf("%d",&secim);
								if(secim==1){
									fclose(magaza);
									isiklandirma();}
						}
					}	
					break;
				case 3:
					remove(dosyauzantisi);
					FILE *magaza = fopen(dosyauzantisi,"w");
					MagazaKiralama();
					fwrite(&Boyut,sizeof(Boyut),1,magaza);
					fclose(magaza);
					YonetimAnasayfa();
				break;
			}
		}
		else{
			printf("\nMevcut kaydınız işleme geçirilememiştir.\n[1] Ana Menüye Dönme\n[2] Kiralama Menüsü\n Giriniz:");
			scanf("%d",&secim);
			if(secim==1)
				AnaMenu();
			else if(secim==2)
				fclose(magaza);
				MagazaKayit();	
		}
		fclose(magaza);
	}
}
